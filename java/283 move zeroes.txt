public class Solution {
    public void moveZeroes(int[] nums) {
        int l = nums.length;
        int left = 0;
        for(int i = 0; i < l; i++) {
            if(nums[i] != 0 ) {
                int tmp = nums[left];
                nums[left] = nums[i];
                nums[i] = tmp;
                left += 1;
            }            
        }
    }
}
